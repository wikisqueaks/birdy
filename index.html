Coniferous vs Deciduous

Means and Error

hab.stat <- bird.dat %>%
  group_by(habitat,replicate) %>%
  summarize(total.abun=sum(abundance)) %>%
  group_by(habitat) %>%
  summarize(mean.abun.hab=mean(total.abun),
            count = n(), sd=sd(total.abun),
            se=sd/sqrt(count))

2x2 Bar Graph -Both Predictors

No statistics tests required for this - just graph / effect size and interpretation of possible interaction effect

factorial.stat <- bird.dat %>%
  group_by(habitat, urban, site, replicate) %>%
  summarise(total.site.abun=sum(abundance)) %>%
  group_by(site,habitat,urban) %>%
  summarise(mean.site.abun=mean(total.site.abun),

            sd=sd(total.site.abun),

# error matrix
error <- factorial.stat %>%
  ungroup %>%
  select(urban,habitat,se) %>%
  pivot_wider(names_from=habitat, values_from=se)

head(error1)
##              C        D
## [1,] 0.3333333 1.154701
## [2,] 0.5773503 3.480102
# Plot

pretty <- c("lightgoldenrod1","lightblue3") # change colour scheme here

factorial.plot <- barplot(factor.stat.box.matrix,
                          beside=TRUE,
                          ylim=c(0,17.5),
                          ylab="Mean Bird Abundance +/- SE",
                          col=pretty,
                          space=c(0,.5),
                          main="Effects of Urban Environment and Tree Type")
# error bars
arrows(factorial.plot,factor.stat.box.matrix+error1,factorial.plot,factor.stat.box.matrix-error1,code=3, angle=90, length=0.1)
# legend
legend(3,15, legend=c("Rural","Urban"), fill=pretty)
